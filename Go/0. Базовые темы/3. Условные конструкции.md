# управляющие конструкции

## if/else

```go
// Основная форма
if x > 0 {
    fmt.Println("положительное")
} else if x < 0 {
    fmt.Println("отрицательное")
} else {
    fmt.Println("ноль")
}

// С инициализацией
if num := getNumber(); num > 100 {
    fmt.Println("больше 100")
}

// Проверка на ошибку
if err := doSomething(); err != nil {
    return err
}
```


## Операторы сравнения

```go
==  // равно
!=  // не равно
<   // меньше
<=  // меньше или равно
>   // больше
>=  // больше или равно
```

## Логические операторы

```go
&&  // И (AND)
||  // ИЛИ (OR)
!   // НЕ (NOT)

// Короткое замыкание
if ptr != nil && ptr.field > 0 {  // ptr.field проверится только если ptr != nil
    // код
}
```

## switch

```go
// По значению
switch day {
case "понедельник":
    fmt.Println("начало недели")
case "суббота", "воскресенье":
    fmt.Println("выходной")
default:
    fmt.Println("рабочий день")
}

// С инициализацией
switch x := getValue(); x {
case 1:
    fmt.Println("один")
case 2:
    fmt.Println("два")
}

// По условию (без переменной)
switch {
case age < 18:
    fmt.Println("несовершеннолетний")
case age < 65:
    fmt.Println("взрослый")
default:
    fmt.Println("пенсионер")
}

// fallthrough - переход к следующему case
switch grade {
case "A":
    fmt.Println("отлично")
    fallthrough
case "B":
    fmt.Println("хорошо или отлично")
}
```

## type switch

```go
// Проверка типа интерфейса
switch v := interface{}(x).(type) {
case int:
    fmt.Printf("целое: %d\n", v)
case string:
    fmt.Printf("строка: %s\n", v)
case bool:
    fmt.Printf("булево: %t\n", v)
default:
    fmt.Printf("неизвестный тип: %T\n", v)
}
```

## Тернарный оператор (альтернативы)

```go
func ternary(condition bool, a, b string) string {
    if condition {
        return a
    }
    return b
}
```
