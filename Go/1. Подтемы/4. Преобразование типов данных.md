# Преобразование типов

## Базовые типы

```go
var i int = 42
var f float64 = float64(i)   // int -> float64
var u uint = uint(f)         // float64 -> uint
var s string = string(i)     // int -> string (ASCII код)
```

## Строки и числа

```go
import "strconv"

// string -> int
i, err := strconv.Atoi("123")
i64, err := strconv.ParseInt("123", 10, 64)

// int -> string  
s := strconv.Itoa(123)
s = strconv.FormatInt(123, 10)

// string -> float
f, err := strconv.ParseFloat("3.14", 64)

// float -> string
s = strconv.FormatFloat(3.14, 'f', 2, 64)
```

## Байты и строки

```go
s := "hello"
b := []byte(s)      // string -> []byte
s = string(b)       // []byte -> string

// Руны
r := []rune(s)      // string -> []rune
s = string(r)       // []rune -> string
```

## Интерфейсы

```go
var i interface{} = 42

// Type assertion
num := i.(int)              // паника если тип не int
num, ok := i.(int)          // безопасно, ok = false если не int

// Type switch
switch v := i.(type) {
case int:
    fmt.Println("int:", v)
case string:
    fmt.Println("string:", v)
default:
    fmt.Println("unknown")
}
```

## Указатели

```go
var i int = 42
p := &i         // *int
val := *p       // разыменование

// Небезопасное преобразование
import "unsafe"
ptr := unsafe.Pointer(&i)
f := (*float32)(ptr)
```

## Структуры

```go
type A struct{ x int }
type B struct{ x int }

a := A{42}
b := B(a)       // преобразование с одинаковой структурой
```
